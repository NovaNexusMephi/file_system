27template <typename T>

52template <typename T>

137template <typename T>
138[[nodiscard]] std::optional<std::vector<T>>
139vector(const libconfig::Setting &from, const std::string &as) {
140 if (!from.exists(as)) {
141 return std::nullopt;
142 }
143
144 try {
145 const auto &values = from.lookup(as);
146 if (values.getType() != libconfig::Setting::TypeArray) {
147 return std::nullopt;
148 }
149
150 std::vector<T> linear_value;
151 linear_value.reserve(values.getLength());
152
153 for (int i = 0; i < values.getLength(); ++i) {
154 T elem;
155 if (!values[i].lookupValue(elem)) {
156 return std::nullopt;
157 }
158 linear_value.push_back(std::move(elem));
159 }
160
161 return linear_value;
162 } catch (const libconfig::SettingException &) {
163 return std::nullopt;
164 }
165}
